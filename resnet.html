class ResNet(nn.Module):
    def __init__(self, res_block_layers, num_classes = 1000):
        super(ResNet, self).__init__()
        layers = OrderedDict([
                ("conv_layer", get_conv_layer(3, res_block_layers[0]['out_channels'] , kernel_size=7, stride=2, padding=3, bias=False))
            ]
        )
        layers["max_pool"] = nn.MaxPool2d(kernel_size = 3, stride = 2, padding = 1)
        in_channels = res_block_layers[0]['out_channels']
        for i, layer in enumerate(res_block_layers, start=1):
            layers[f"res_layer_{i}"] = OrderedDict()
            res_blocks = OrderedDict()
            res_blocks[f"res_block_{i}_1"] = ResidualBlock(in_channels, layer['out_channels'], stride=layer['stride'])
            for j in range(2, layer["num_blocks"]+1):
                res_blocks[f"res_block_{i}_{j}"] = ResidualBlock(layer["out_channels"], layer['out_channels'], stride=1)

            layers[f"res_layer_{i}"] = nn.Sequential(res_blocks)
            in_channels = layer["out_channels"]
            
        layers['avgpool'] = nn.AdaptiveAvgPool2d(output_size=(1, 1))
        layers['flatten'] = nn.Flatten()
        layers['fc'] = nn.Linear(in_channels, num_classes)
        
        self.model = nn.Sequential(layers)
    
    def forward(self, x):
        return self.model(x)

layers_lst = list()
def get_children(layer, level=0):
    global layers_lst
    named_children = list(layer.named_children())
    for name, child in named_children:
        is_leaf_node = len(list(child.named_children())) == 0
        layers_lst.append({"name": name, "layer": child, "level": level, "is_leaf_node": is_leaf_node})
        get_children(child, level+1)
